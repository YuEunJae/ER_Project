CREATE TABLE PHOTOREVIEW(
    PNO BIGINT(10) AUTO_INCREMENT PRIMARY KEY,
    PCODE VARCHAR(30),
    PHOTOIMAGE VARCHAR(100),
    PHOTOWRITER VARCHAR(50),
    PHOTOTITLE VARCHAR(100),
    PHOTOSTAR BIGINT(20),
    PHOTOCONTENT VARCHAR(1000),
    created varchar(100)
    
);

select p.PNO as pno,
		         p.PCODE as pcode,
		         p.PHOTOIMAGE as photoimage,
		         p.PHOTOWRITER as photowriter ,
		         p.PHOTOTITLE as phototitle,
		         p.PHOTOSTAR as photostar,
		         p.PHOTOCONTENT as photocontent,
		         p.CREATED as created
		 from (select PNO,
			         PCODE,
			         PHOTOIMAGE,
			         PHOTOWRITER,
			         PHOTOTITLE,
			         PHOTOSTAR,
			         PHOTOCONTENT,
		             created
		        from PHOTOREVIEW 
			  			where PCODE = 'ALTON'  
		    order by PNO DESC ) as p;
